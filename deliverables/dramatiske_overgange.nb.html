##Find frste stning i hver scene. Hjlp fra Asger:

first_sentence <- play %>%
  #Group by s vi hndterer hver scene for sig
 group_by(act_number, scene_number) %>%
  #Filtrer alle linier som ikke er replikker fra
 filter(!is.na(spoke)) %>%
  # Sorter i hver gruppe, s linierne kommer i rkkeflge
 arrange(index, .by_group = TRUE) %>%
  # Lav et ny index i hver gruppe, startende med den frste.
  # P den mde ved jeg at den frste i hver gruppe er nr 1.
 mutate(group_row_num = row_number()) %>%
  # Kun behold den frste i hver gruppe
 filter(group_row_num == 1) %>%
  # Fjern gruppe index for det er ikke relevant lngere
 select(- group_row_num )

Mon ikke vi kan gre noget tilsvarende for at finde den sidste stning i hver scene? Hjlp fra Asger:

last_sentence <- play %>%
  #Group by s vi hndterer hver scene for sig
 group_by(act_number, scene_number) %>%
  #Filtrer alle linier som ikke er replikker fra
 filter(!is.na(spoke)) %>%
  # Sorter i hver gruppe, s linierne kommer i omvendt rkkeflge
 arrange(desc(index), .by_group = TRUE) %>%
  # Lav et ny index i hver gruppe, startende med den sidste.
  # P den mde ved jeg at den sidste i hver gruppe er nr 1.
 mutate(group_row_num = row_number()) %>%
  # Kun behold den sidste i hver gruppe
 filter(group_row_num == 1) %>%
  # Fjern gruppe index for det er ikke relevant lngere
 select(- group_row_num )

write.csv(last_sentence, file = here("test-data/last_sentence.csv"))
